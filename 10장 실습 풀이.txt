#include <iostream>
#include <vector>
using namespace std;

int main()
{
	vector<int> vi;
	
	while (true)
	{
		cout << "정수를 입력(0을 입력하면 종료)>> ";
		int n;
		cin >> n;
		if (n == 0)
			break;
		vi.push_back(n);
		
		int sum = 0;
		for (auto it = vi.begin();it != vi.end();it++)
		{
			cout << *it << ' ';
			sum += *it;
		}
		cout << "\n평균 = " << (double)sum / vi.size() << endl;
	}
}

#include <iostream>
#include <string>
#include <vector>
using namespace std;

class Book {
	int year;
	string title, author;
public:
	Book(int y, string t, string a) :year(y), title(t), author(a){}
	string getAuthor() { return author; }
	int getYear() { return year; }
	void show() { cout << year << "년도, " << title << ", " << author << endl; }
};

int main()
{
	vector<Book> lib;
	int year;
	string title, author;

	cout << "입고할 책을 입력(년도에 -1을 입력하면 입고 종료)\n";
	while (true)
	{
		cout << "년도>> ";
		cin >> year;
		if (year == -1)
			break;
		cin.ignore(10, '\n');
		cout << "책이름>> ";
		getline(cin, title);
		cout << "저자>> ";
		getline(cin, author);
		lib.push_back(Book(year,title,author));
	}
	
	cout << "총 입고된 책은 " << lib.size() << "권 입니다." << endl;
	cin.ignore(10, '\n');
	cout << "검색하고자 하는 저자 입력>> ";
	getline(cin, author);
	cout << "검색하고자 하는 년도 입력>> ";
	cin >> year;
	
	for (auto it = lib.begin();it != lib.end();it++)
	{
		if (it->getAuthor() == author && it->getYear() == year)
			it->show();
	}
}

#include <iostream>
#include <string>
#include <map>
using namespace std;

int main()
{
	map<string, int> list;
	string name;
	int score;

	cout << "**** 점수관리 프로그램 HIGH SCORE를 시작합니다 ****" << endl;
	while (true)
	{
		int menu;
		cout << "입력:1, 조회:2, 종료:3 >> ";
		cin >> menu;
		if (menu == 3)
			break;
		switch (menu)
		{
		case 1:
			cout << "이름과 점수 >> ";
			cin >> name >> score;
			list[name] = score;
			break;
		case 2:
			cout << "이름 >> ";
			cin >> name;
			for (auto it = list.begin();it != list.end();it++)
			{
				if (it->first == name)
				{
					cout << it->first << "의 점수는 " << it->second << endl;
					break;
				}
			}
			break;
		default:
			cout << "잘못된 입력입니다." << endl;
			break;
		}
	}
	cout << "프로그램을 종료합니다." << endl;
}
