<1번>
#include <iostream>
using namespace std;

int main()
{
	int n;
	while (true)
	{
		try {
			cout << "양수 입력 >> ";
			cin >> n;

			if (cin.fail())
				throw"숫자 이외의 입력 발생.";
			else if (n < 1 || n>9)
				throw n;
			for (int i = 1;i < 10;i++)
				cout << n << '*' << i << '=' << n * i << ' ';
			cout << endl;
		}
		catch (const char* s)
		{
			cout << s << "입력 오류가 발생하여 더 이상 입력되지 않습니다. 프로그램을 종료합니다." << endl;
			return 0;
		}
		catch (int n)
		{
			cout << n << "은 올바른 숫자가 아닙니다. 1~9 사이의 정수를 입력해주세요." << endl;
		}
	}
}
