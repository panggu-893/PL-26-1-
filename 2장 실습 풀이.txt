#include <iostream>
using namespace std;

int main()
{
	double arr[6] = { 0 };
	double max, min;
	cout << "실수 6개 입력>> ";
	cin >> arr[0];
	max = min = arr[0];
	for (int i = 1;i < 6;i++)
	{
		cin >> arr[i];
		if (arr[i] > max) max = arr[i];
		if (arr[i] < min) min = arr[i];
	}

	cout << "가장 큰 수 : " << max << endl;
	cout << "가장 작은 수 : " << min << endl;

	return 0;
}

#include <iostream>
using namespace std;

int main()
{
	cout << "**중국집**" << endl;
	int flag = 1;
	while (flag)
	{
		int menu;
		cout << "짬뽕:1, 짜장면:2, 탕수육:3, 종료:4>>";
		cin >> menu;
		switch (menu)
		{
			int count;
		case 1:
			cout << "몇인분? >>";
			cin >> count;
			cout << "짬뽕 " << count << "인분~" << endl;
			break;
		case 2:
			cout << "몇인분? >>";
			cin >> count;
			cout << "짜장면 " << count << "인분~" << endl;
			break;
		case 3:
			cout << "몇인분? >>";
			cin >> count;
			cout << "탕수육 " << count << "인분~" << endl;
			break;
		case 4:
			flag = 0;
			break;
		}
	}
	cout << "영업 종료 감사합니다." << endl;

	return 0;
}
