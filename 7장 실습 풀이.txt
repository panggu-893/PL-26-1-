#include <iostream>
#include <string>
using namespace std;

class Book {
	string title;
	int price;
	int pages;
public:
	Book(string t="",int pr=0,int pa=0):title(t),price(pr),pages(pa){}
	void show() { cout << title << ' ' << price << "원 " << pages << " 페이지" << endl; }
	string getTitle() { return title; }
	void operator +=(int add) { price += add; return; }
	void operator -=(int subs) { price -= subs;  return; }
};

int main()
{
	Book a("청춘", 20000, 300), b("미래", 30000, 500);
	a += 500;
	b -= 500;
	a.show();
	b.show();
}

#include <iostream>
using namespace std;

class Stack {
	int size;
	int* data;
	int tos;
public:
	Stack(int size = 10) { this->size = size; data = new int[size]; tos = 0; }
	~Stack() { delete[]data; }
	Stack& operator <<(int n);
	void operator >>(int& p);
	bool operator!();
};

Stack& Stack::operator<<(int n)
{
	if (tos >= size)
		return *this;
	else
	{
		data[tos++] = n;
		return *this;
	}
}

void Stack::operator>>(int& p)
{
	if (tos <= 0)
		return;
	else
		p = data[--tos];
}

bool Stack::operator!()
{
	if (tos <= 0)
		return true;
	else
		return false;
}

int main()
{
	Stack stack;
	stack << 3 << 5 << 10;
	while (true)
	{
		if (!stack) break;
		int x;
		stack >> x;
		cout << x << ' ';
	}
	cout << endl;
}
